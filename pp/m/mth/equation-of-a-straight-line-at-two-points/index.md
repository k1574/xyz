# Equation of a straight line at two points

	f(x1) = y1 ;
	f(x2) = y2 ;

	f = x*(y2-y1)/(x2-x1) + y1 - x1*(y2-y1)/(x2-x1) ;

	▲
		f = k*x + c ;
		k = y/x = (y2-y1)/(x2-x1) ;

		y1 = k*x1 + c = (y2-y1)/(x2-x1)*x1 + c ;
		c = y1 - x1*(y2-y1)*(x2-x1) ;

		f = x*(y2-y1)/(x2-x1) + y1 - x1*(y2-y1)/(x2-x1) ;
	∎
